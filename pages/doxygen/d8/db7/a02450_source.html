61 bool OnGossipSelect(Player* player, Creature* creature, uint32 /*sender*/, uint32 action) override

66 case GOSSIP_ACTION_INFO_DEF+1:

69 break;
70 case GOSSIP_ACTION_INFO_DEF+11:
71 CloseGossipMenuFor(player);
72 player->CastSpell(player, SPELL_LEARN_SMELT, false);
73 break;
74 case GOSSIP_ACTION_INFO_DEF+2:
75 AddGossipItemFor(player, GOSSIP_TEXT_CONTINUE, 1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 22);
76 SendGossipMenuFor(player, SAY_QUEST_ACCEPTED, creature->GetGUID());
77 break;
78 case GOSSIP_ACTION_INFO_DEF+22:
79 CloseGossipMenuFor(player);
80 if (InstanceScript* instance = creature->GetInstanceScript())
81 {
82 //are 5 minutes expected? go template may have data to despawn when used at quest
83 instance->DoRespawnGameObject(instance->GetGuidData(DATA_GO_CHALICE), MINUTE * 5);
84 }
85 break;
86 }
87 return true;
88 }
89
90 bool OnGossipHello(Player* player, Creature* creature) override
91 {
92 if (player->GetQuestRewardStatus(QUEST_SPECTRAL_CHALICE) == 1 && player->GetSkillValue(SKILL_MINING) >= DATA_SKILLPOINT_MIN && !player->HasSpell(SPELL_SMELT_DARK_IRON))
93 {
94 AddGossipItemFor(player, GOSSIP_GROOMREL, 0, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 1);
95 SendGossipMenuFor(player, SAY_QUEST_COMPLETED, creature->GetGUID());
96 }
97
98 if (player->GetQuestRewardStatus(QUEST_SPECTRAL_CHALICE) == 0 && player->GetSkillValue(SKILL_MINING) >= DATA_SKILLPOINT_MIN)
99 AddGossipItemFor(player, GOSSIP_GROOMREL, 1, GOSSIP_SENDER_MAIN, GOSSIP_ACTION_INFO_DEF + 2);
100
101 SendGossipMenuFor(player, player->GetGossipTextId(creature), creature->GetGUID());
102 return true;
103 }

176 void Reset() override
177 {
178 Voidwalkers = false;
179 me->SetFaction(FACTION_FRIENDLY);
180
181 // was set before event start, so set again
182 me->SetImmuneToPC(true);
183
184 if (instance->GetData(TYPE_TOMB_OF_SEVEN) == DONE) // what is this trying to do? Probably some kind of crash recovery
185 {
186 me->ReplaceAllNpcFlags(UNIT_NPC_FLAG_NONE);

190 me->ReplaceAllNpcFlags(UNIT_NPC_FLAG_GOSSIP);
191 }
192 }

195 {
196 _events.ScheduleEvent(EVENT_SPELL_SHADOWBOLTVOLLEY, 10000);
197 _events.ScheduleEvent(EVENT_SPELL_IMMOLATE, 18000);
198 _events.ScheduleEvent(EVENT_SPELL_CURSEOFWEAKNESS, 5000);
199 _events.ScheduleEvent(EVENT_SPELL_DEMONARMOR, 16000);
200 _events.ScheduleEvent(EVENT_SPELL_SUMMON_VOIDWALKERS, 1000);
201 }
202
203 void EnterEvadeMode(EvadeReason /*why*/) override
204 {
205 me->RemoveAllAuras();

207 me->CombatStop(true);
208 me->LoadCreaturesAddon(true);
209 if (me->IsAlive())
210 me->GetMotionMaster()->MoveTargetedHome();

214 void UpdateAI(uint32 diff) override
215 {
216 if (!UpdateVictim())
217 return;
218
219 _events.Update(diff);
220
221 switch(_events.ExecuteEvent())
222 {
223 case EVENT_SPELL_SHADOWBOLTVOLLEY:
224 DoCastVictim(SPELL_SHADOWBOLTVOLLEY);
225 _events.ScheduleEvent(EVENT_SPELL_SHADOWBOLTVOLLEY, 12000);
226 break;
227 case EVENT_SPELL_IMMOLATE:
228 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 100, true))
229 {
230 DoCast(target, SPELL_IMMOLATE);
231 _events.ScheduleEvent(EVENT_SPELL_IMMOLATE, 25000);
232 }
233 // Didn't get a target, try again in 1s
234 _events.ScheduleEvent(EVENT_SPELL_IMMOLATE, 1000);
235 break;
236 case EVENT_SPELL_CURSEOFWEAKNESS:
237 DoCastVictim(SPELL_CURSEOFWEAKNESS);
238 _events.ScheduleEvent(EVENT_SPELL_CURSEOFWEAKNESS, 45000);
239 break;
240 case EVENT_SPELL_DEMONARMOR:
241 DoCast(me, SPELL_DEMONARMOR);
242 _events.ScheduleEvent(EVENT_SPELL_DEMONARMOR, 300000);
243 break;
244 case EVENT_SPELL_SUMMON_VOIDWALKERS:
245 if (!Voidwalkers && HealthBelowPct(51))
246 {
247 DoCastVictim(SPELL_SUMMON_VOIDWALKERS, true);
248 Voidwalkers = true;
249 }
250 // Not ready yet, try again in 1s
251 _events.ScheduleEvent(EVENT_SPELL_SUMMON_VOIDWALKERS, 1000);
252 break;
253 }
254
255 DoMeleeAttackIfReady();
256 }
257 };
