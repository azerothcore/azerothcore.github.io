110 void UpdateAI(uint32 diff) override
111 {
112 // Return since we have no target
113 if (!UpdateVictim())
114 return;
115
116 _events.Update(diff);
117
118 if (me->HasUnitState(UNIT_STATE_CASTING))
119 return;
120
121 while (uint32 eventId = _events.ExecuteEvent())
122 {
123 switch (eventId)
124 {
125 case EVENT_SHADOW_VOLLEY:
126 DoCastVictim(SPELL_SHADOW_VOLLEY);
127 _events.ScheduleEvent(EVENT_SHADOW_VOLLEY, urand(4000, 6000));
128 break;
129 case EVENT_CLEAVE:
130 DoCastVictim(SPELL_CLEAVE);
131 _events.ScheduleEvent(EVENT_CLEAVE, urand(8000, 12000));
132 break;
133 case EVENT_THUNDERCLAP:
134 DoCastVictim(SPELL_THUNDERCLAP);
135 _events.ScheduleEvent(EVENT_THUNDERCLAP, urand(10000, 14000));
136 break;
137 case EVENT_VOID_BOLT:
138 DoCastVictim(SPELL_VOID_BOLT);
139 _events.ScheduleEvent(EVENT_VOID_BOLT, urand(15000, 18000));
140 break;
141 case EVENT_MARK_OF_KAZZAK:
142 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, PowerUsersSelector(me, POWER_MANA, 100.0f, true)))
143 DoCast(target, SPELL_MARK_OF_KAZZAK);
144 _events.ScheduleEvent(EVENT_MARK_OF_KAZZAK, 20000);
145 break;
146 case EVENT_ENRAGE:
147 Talk(EMOTE_FRENZY);
148 DoCast(me, SPELL_ENRAGE);
149 _events.ScheduleEvent(EVENT_ENRAGE, 30000);
150 break;
151 case EVENT_TWISTED_REFLECTION:
152 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.0f, true))
153 DoCast(target, SPELL_TWISTED_REFLECTION);
154 _events.ScheduleEvent(EVENT_TWISTED_REFLECTION, 15000);
155 break;
156 case EVENT_BERSERK:
157 DoCast(me, SPELL_BERSERK);
158 break;
159 default:
160 break;
161 }
162 }
163
164 DoMeleeAttackIfReady();
165 }
166
167 private:
168 EventMap _events;
169 };
