93 void MovementInform(uint32 type, uint32 point) override
94 {
95 if (type != POINT_MOTION_TYPE && type != EFFECT_MOTION_TYPE)
96 return;
97
98 switch (point)
99 {
100 case POINT_FLIGHT:
101 me->SetFacingTo(4.69f);
102 events.ScheduleEvent(EVENT_CONFLAGRATION, 1000);
103 events.ScheduleEvent(EVENT_LAND_BACK, 7000);
104 Talk(SAY_CONFLAGRATION);
105 break;
106 case POINT_LAND:
107 me->SetDisableGravity(false);
108 events.ScheduleEvent(EVENT_LAND_GROUND, 500);
109 break;
110 }
111 }
112
113 void JustReachedHome() override
114 {
115 BossAI::JustReachedHome();
116 me->SetDisableGravity(false);
117 me->SetHover(false);
118 }
119
120 void KilledUnit(Unit* /*victim*/) override
121 {
122 if (events.GetNextEventTime(EVENT_KILL_TALK) == 0)
123 {
124 Talk(SAY_KILL);
125 events.ScheduleEvent(EVENT_KILL_TALK, 6000);
126 }
127 }
128
129 void UpdateAI(uint32 diff) override
130 {
131 if (!UpdateVictim())
132 return;
133
134 events.Update(diff);
135 if (me->HasUnitState(UNIT_STATE_CASTING))
136 return;
137
138 switch (events.ExecuteEvent())
139 {
140 case EVENT_FLIGHT:
141 {
142 me->SetReactState(REACT_PASSIVE);
143 me->AttackStop();
144 me->SetDisableGravity(true);
145 me->GetMotionMaster()->MovePoint(POINT_TAKEOFF, me->GetPositionX(), me->GetPositionY(), me->GetPositionZ() + 6.0f, false);
146 events.ScheduleEvent(EVENT_FLIGHT, 50000);
147 events.DelayEvents(15000);
148 events.ScheduleEvent(EVENT_AIR_MOVEMENT, 2000);
149 break;
150 }
151 case EVENT_CONFLAGRATION:
152 me->CastCustomSpell(SPELL_CONFLAGRATION, SPELLVALUE_MAX_TARGETS, RAID_MODE(3, 6, 3, 6), me, true);
153 break;
154 case EVENT_ENRAGE:
155 me->CastSpell(me, SPELL_ENRAGE, false);
156 Talk(EMOTE_ENRAGED);
157 events.ScheduleEvent(EVENT_ENRAGE, urand(15000, 20000));
158 break;
159 case EVENT_FLAME_BREATH:
160 me->CastSpell(me->GetVictim(), SPELL_FLAME_BREATH, false);
161 events.ScheduleEvent(EVENT_FLAME_BREATH, urand(20000, 30000));
162 break;
163 case EVENT_AIR_MOVEMENT:
164 me->GetMotionMaster()->MovePoint(POINT_FLIGHT, 3155.51f, 683.844f, 95.0f, false);
165 break;
166 case EVENT_LAND_BACK:
167 me->GetMotionMaster()->MovePoint(POINT_LAND, 3151.07f, 636.443f, 80.0f, false);
168 break;
169 case EVENT_LAND_GROUND:
170 me->SetReactState(REACT_AGGRESSIVE);
171 if (me->GetVictim())
172 me->GetMotionMaster()->MoveChase(me->GetVictim());
173 break;
174 }
175
176 DoMeleeAttackIfReady();
177 }
178 };
