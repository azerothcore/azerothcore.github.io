116 void ProcessSlaughterEvent()
117 {
118 if (_slaughterProgress == 1)
119 {
120 if (Creature* baron = instance->GetCreature(_baronRivendareGUID))
121 baron->AI()->Talk(SAY_BRAON_SUMMON_RAMSTEIN);
122
123 Position pos = {4032.20f, -3378.06f, 119.75f, 4.67f};
124 instance->SummonCreature(NPC_RAMSTEIN_THE_GORGER, pos);
125 }
126 if (_slaughterProgress == 2)
127 {
128 for (uint32 i = 0; i < 33; ++i)
129 events.ScheduleEvent(EVENT_SPAWN_MINDLESS, 5000 + i * 210);
130 if (Creature* baron = instance->GetCreature(_baronRivendareGUID))
131 if (GameObject* gate = baron->FindNearestGameObject(GO_SLAUGHTER_GATE_SIDE, 200.0f))
132 gate->SetGoState(GO_STATE_ACTIVE);
133 }
134 if (_slaughterProgress == 3)
135 {
136 events.ScheduleEvent(EVENT_SPAWN_BLACK_GUARD, 20000);
137 }
138 if (_slaughterProgress == 4)
139 {
140 if (Creature* baron = instance->GetCreature(_baronRivendareGUID))

142 if (GameObject* gate = instance->GetGameObject(_zigguratDoorsGUID5))

262 void DoSpawnPlaguedCritters(uint8 /*uiGate*/, Player* player)
263 {
264 if (!player)
265 return;
266
267 uint32 uiEntry = aPlaguedCritters[urand(0, 2)];
268 for (uint8 i = 0; i < 30; ++i)
269 {
270 float x, y, z;
271 const Position pPos = { player->GetPositionX(), player->GetPositionY(), player->GetPositionZ(), player->GetOrientation() };
272 player->GetRandomPoint(pPos, 8.0f, x, y, z);
273 z = player->GetPositionZ() + 1;
274 player->SummonCreature(uiEntry, x, y, z, 0, TEMPSUMMON_DEAD_DESPAWN, 0)->AI()->AttackStart(player);
275 }

396 void Update(uint32 diff) override
397 {
398 events.Update(diff);
399
400 Map::PlayerList const& players = instance->GetPlayers();
401 // Loop over the two Gate traps, each one has up to three timers (trap reset, gate opening delay, critters spawning delay)
402 for (uint8 i = 0; i < 2; i++)
403 {
404 // if the gate is in cooldown, skip the other checks
405 if (_gateTrapsCooldown[i])
406 break;
407
408 // Check that the trap is not on cooldown, if so check if player/pet is in range
409 for (Map::PlayerList::const_iterator itr = players.begin(); itr != players.end(); ++itr)
410 {
411 if (Player* player = itr->GetSource())
412 {
413 // should pet also trigger the trap? could not find any source for it
414 if (!player->IsGameMaster() && player->IsWithinDist2d(aGateTrap[i].m_positionX, aGateTrap[i].m_positionY, 5.5f))
415 {
416 // Check if timer was not already set by another player/pet a few milliseconds before
417 if (_gateTrapsCooldown[i])
418 return;
419
420 _gateTrapsCooldown[i] = true;
421
422 // close the gates
423 if (_trapGatesGUIDs[2 * i])
424 DoUseDoorOrButton(_trapGatesGUIDs[2 * i]);
425 if (_trapGatesGUIDs[2 * i + 1])
426 DoUseDoorOrButton(_trapGatesGUIDs[2 * i + 1]);
427
428 _trappedPlayerGUID = player->GetGUID();
429
430 if (i == 0)
431 {
432 // set timer to reset the trap
433 events.ScheduleEvent(EVENT_GATE1_TRAP, 30 * MINUTE * IN_MILLISECONDS);
434 // set timer to reopen gates
435 events.ScheduleEvent(EVENT_GATE1_DELAY, 20 * IN_MILLISECONDS);
436 // set timer to spawn the plagued critters
437 events.ScheduleEvent(EVENT_GATE1_CRITTER_DELAY, 2 * IN_MILLISECONDS);
438 }
439 else if (i == 1)
440 {
441 // set timer to reset the trap
442 events.ScheduleEvent(EVENT_GATE2_TRAP, 30 * MINUTE * IN_MILLISECONDS);
443 // set timer to reopen gates
444 events.ScheduleEvent(EVENT_GATE2_DELAY, 20 * IN_MILLISECONDS);
445 // set timer to spawn the plagued critters
446 events.ScheduleEvent(EVENT_GATE2_CRITTER_DELAY, 2 * IN_MILLISECONDS);
447 }
448 }
449 }
450 }
451 }

453 const int GATE1 = 0;
454 const int GATE2 = 1;
455
456 switch (events.ExecuteEvent())
457 {
458 case EVENT_GATE1_TRAP:
459 _gateTrapsCooldown[GATE1] = false;
460 break;
461 case EVENT_GATE2_TRAP:

472 break;
473 case EVENT_GATE2_CRITTER_DELAY:
474 gate_critter_delay(GATE2);
475 break;
476 case EVENT_BARON_TIME:
477 {
478 --_baronRunTime;
479 instance->LoadGrid(4035.83f, -3336.31f);
480 Creature* baron = instance->GetCreature(_baronRivendareGUID);
481 if (baron && !baron->IsInCombat())
482 {
483 switch (_baronRunTime)
484 {
485 case 10:
486 baron->AI()->Talk(SAY_BARON_10M);
487 break;
488 case 5:
489 baron->AI()->Talk(SAY_BARON_5M);
490 if (Creature* ysida = baron->FindNearestCreature(NPC_YSIDA, 50.0f))
491 ysida->AI()->SetData(1, 1);
492 break;
493 case 0:
494 baron->AI()->Talk(SAY_BARON_0M);
495 DoRemoveAurasDueToSpellOnPlayers(SPELL_BARON_ULTIMATUM);
496 break;
497 }
498 }
499
500 if (_baronRunTime > 0)
501 events.ScheduleEvent(EVENT_BARON_TIME, 60000);
502 else
503 events.ScheduleEvent(EVENT_EXECUTE_PRISONER, 0);
504
505 SaveToDB();
506 break;
507 }
508 case EVENT_EXECUTE_PRISONER:
509 {
510 instance->LoadGrid(4035.83f, -3336.31f);
511 Creature* baron = instance->GetCreature(_baronRivendareGUID);
512 if (baron && baron->IsAlive())
513 {
514 if (!baron->IsInCombat())
515 {
516 baron->HandleEmoteCommand(EMOTE_ONESHOT_ATTACK1H);
517 if (Creature* ysida = baron->FindNearestCreature(NPC_YSIDA, 50.0f))
518 Unit::Kill(baron, ysida);
519 }
520 else
521 events.ScheduleEvent(EVENT_EXECUTE_PRISONER, 1000);
522 }

526 {
527 Position pos = {3941.75f, -3393.06f, 119.70f, 0.0f};
528 instance->SummonCreature(NPC_MINDLESS_UNDEAD, pos);

532 {
533 Map::PlayerList const& PlayerList = instance->GetPlayers();
534 if (!PlayerList.IsEmpty())
535 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
536 if (Player* player = i->GetSource())
537 if (player->GetDistance2d(4034.97f, -3402.13f) < 50.0f)

549 if (Creature* guard = instance->SummonCreature(NPC_BLACK_GUARD, BlackGuardPos[i]))

556 }
557
558 if (GameObject* gate = instance->GetGameObject(_zigguratDoorsGUID4))
559 gate->SetGoState(GO_STATE_ACTIVE);
560 break;
561 }
