28template<typename T> // requires AsyncCallback<T>
29class AsyncCallbackProcessor
30{
31public:
32 AsyncCallbackProcessor() = default;
33 ~AsyncCallbackProcessor() = default;
34
35 T& AddCallback(T&& query)
36 {
37 _callbacks.emplace_back(std::move(query));
38 return _callbacks.back();
39 }
40
41 void ProcessReadyCallbacks()
42 {
43 if (_callbacks.empty())
44 return;
45
46 std::vector<T> updateCallbacks{ std::move(_callbacks) };
47
48 updateCallbacks.erase(std::remove_if(updateCallbacks.begin(), updateCallbacks.end(), [](T& callback)
49 {
50 return callback.InvokeIfReady();
51 }), updateCallbacks.end());
52
53 _callbacks.insert(_callbacks.end(), std::make_move_iterator(updateCallbacks.begin()), std::make_move_iterator(updateCallbacks.end()));
54 }
55
56private:
57 AsyncCallbackProcessor(AsyncCallbackProcessor const&) = delete;
58 AsyncCallbackProcessor& operator=(AsyncCallbackProcessor const&) = delete;
59
60 std::vector<T> _callbacks;
61};
