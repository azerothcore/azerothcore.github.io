137 void UpdateAI(uint32 diff) override
138 {
139 if (!UpdateVictim())
140 return;
141
142 if (WhirlwindTimer <= diff)
143 {
144 DoCastVictim(SPELL_WHIRLWIND);
145 WhirlwindTimer = urand(8 * IN_MILLISECONDS, 18 * IN_MILLISECONDS);
146 }
147 else WhirlwindTimer -= diff;
148
149 if (Whirlwind2Timer <= diff)
150 {
151 DoCastVictim(SPELL_WHIRLWIND2);
152 Whirlwind2Timer = urand(7 * IN_MILLISECONDS, 25 * IN_MILLISECONDS);
153 }
154 else Whirlwind2Timer -= diff;
155
156 if (KnockdownTimer <= diff)
157 {
158 DoCastVictim(SPELL_KNOCKDOWN);
159 KnockdownTimer = urand(10 * IN_MILLISECONDS, 15 * IN_MILLISECONDS);
160 }
161 else KnockdownTimer -= diff;
162
163 if (FrenzyTimer <= diff)
164 {
165 DoCastVictim(SPELL_FRENZY);
166 FrenzyTimer = urand(20 * IN_MILLISECONDS, 30 * IN_MILLISECONDS);
167 }
168 else FrenzyTimer -= diff;
169
170 if (YellTimer <= diff)
171 {
172 Talk(YELL_RANDOM);
173 YellTimer = urand(20 * IN_MILLISECONDS, 30 * IN_MILLISECONDS); //20 to 30 seconds
174 }
175 else YellTimer -= diff;
176
177 // check if creature is not outside of building
178 if (ResetTimer <= diff)
179 {
180 if (me->GetDistance2d(me->GetHomePosition().GetPositionX(), me->GetHomePosition().GetPositionY()) > 50)
181 {
182 ScriptedAI::EnterEvadeMode();
183 Talk(YELL_EVADE);
184 }
185 ResetTimer = 5 * IN_MILLISECONDS;
186 }
187 else ResetTimer -= diff;
188
189 DoMeleeAttackIfReady();
190 }
191 };
