284 icewall->SetGoState(GO_STATE_ACTIVE);

287 m_auiEncounter[2] = data;

393 void Load(const char* in) override
394 {
395 if (!in)
396 {
397 OUT_LOAD_INST_DATA_FAIL;
398 return;
399 }
400
401 OUT_LOAD_INST_DATA(in);
402
403 char dataHead1, dataHead2;
404 uint32 data0, data1, data2, data3;
405
406 std::istringstream loadStream(in);
407 loadStream >> dataHead1 >> dataHead2 >> data0 >> data1 >> data2 >> data3;
408
409 if (dataHead1 == 'P' && dataHead2 == 'S')
410 {
411 m_auiEncounter[0] = data0;
412 m_auiEncounter[1] = data1;
413 m_auiEncounter[2] = data2;
414 InstanceProgress = data3;
415
416 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
417 if (m_auiEncounter[i] == IN_PROGRESS)
418 m_auiEncounter[i] = NOT_STARTED;
419 }
420 else OUT_LOAD_INST_DATA_FAIL;
421
422 OUT_LOAD_INST_DATA_COMPLETE;
423 }
