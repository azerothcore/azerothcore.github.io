130 void UpdateAI(uint32 diff) override
131 {
132 events2.Update(diff);
133 switch (events2.ExecuteEvent())
134 {
135 case EVENT_TELEPORT_VISUAL:
136 me->CastSpell(me, SPELL_TELEPORT_VISUAL, true);
137 break;
138 case EVENT_MILLHOUSE_INTRO1:
139 Talk(SAY_INTRO_1);
140 events2.ScheduleEvent(EVENT_MILLHOUSE_INTRO2, 18000);
141 break;
142 case EVENT_MILLHOUSE_INTRO2:
143 Talk(SAY_INTRO_2);
144 events2.ScheduleEvent(EVENT_MILLHOUSE_INTRO3, 8000);
145 break;
146 case EVENT_MILLHOUSE_INTRO3:
147 Talk(SAY_INTRO_3);
148 events2.ScheduleEvent(EVENT_MILLHOUSE_INTRO4, 6000);
149 break;
150 case EVENT_MILLHOUSE_INTRO4:
151 Talk(SAY_INTRO_4);
152 events2.ScheduleEvent(EVENT_MILLHOUSE_INTRO5, 8000);
153 break;
154 case EVENT_MILLHOUSE_INTRO5:
155 Talk(SAY_WATER);
156 me->CastSpell(me, SPELL_CONJURE_WATER, false);
157 events2.ScheduleEvent(EVENT_MILLHOUSE_INTRO6, 7000);
158 break;
159 case EVENT_MILLHOUSE_INTRO6:
160 Talk(SAY_BUFFS);
161 me->CastSpell(me, SPELL_ICE_ARMOR, false);
162 events2.ScheduleEvent(EVENT_MILLHOUSE_INTRO7, 7000);
163 break;
164 case EVENT_MILLHOUSE_INTRO7:
165 Talk(SAY_DRINK);
166 me->CastSpell(me, SPELL_ARCANE_INTELLECT, false);
167 events2.ScheduleEvent(EVENT_MILLHOUSE_INTRO8, 7000);
168 break;
169 case EVENT_MILLHOUSE_INTRO8:
170 Talk(SAY_READY);
171 me->GetMotionMaster()->MovePoint(1, 445.82f, -158.38f, 43.067f);
172 events2.ScheduleEvent(EVENT_MILLHOUSE_INTRO9, 5000);
173 break;
174 case EVENT_MILLHOUSE_INTRO9:
175 me->SetFacingTo(M_PI * 1.5f);
176 me->SetHomePosition(me->GetPositionX(), me->GetPositionY(), me->GetPositionZ(), M_PI * 1.5f);
177 me->SetImmuneToAll(false);
178 me->SetReactState(REACT_AGGRESSIVE);
179 events2.ScheduleEvent(EVENT_SEARCH_FIGHT, 1000);
180 break;
181 case EVENT_SEARCH_FIGHT:
182 if (!me->IsInCombat() && !me->IsInEvadeMode())
183 if (Unit* target = me->SelectNearbyTarget(nullptr, 30.0f))
184 AttackStart(target);
185 events2.ScheduleEvent(EVENT_SEARCH_FIGHT, 1000);
186 break;
187 }
188
189 if (!UpdateVictim())
190 return;
191
192 events.Update(diff);
193 if (me->HasUnitState(UNIT_STATE_CASTING))
194 return;
195
196 switch (events.ExecuteEvent())
197 {
198 case EVENT_MILL_CHECK_HEALTH:
199 if (HealthBelowPct(20))
200 {
201 Talk(SAY_LOWHP);
202 break;
203 }
204 events.ScheduleEvent(EVENT_MILL_CHECK_HEALTH, 1000);
205 break;
206 case EVENT_MILL_PYROBLAST:
207 Talk(SAY_PYRO);
208 me->CastSpell(me->GetVictim(), SPELL_PYROBLAST, false);
209 events.ScheduleEvent(EVENT_MILL_PYROBLAST, 30000);
210 break;
211 case EVENT_MILL_ICEBLOCK:
212 if (me->GetDistance(me->GetVictim()) < 5.0f)
213 {
214 Talk(SAY_ICEBLOCK);
215 me->CastSpell(me, SPELL_ICEBLOCK, true);
216 break;
217 }
218 events.ScheduleEvent(EVENT_MILL_ICEBLOCK, 1000);
219 break;
220 case EVENT_MILL_BASE_SPELL:
221 switch (RAND(SPELL_FIREBALL, SPELL_ARCANE_MISSILES, SPELL_FROSTBOLT))
222 {
223 case SPELL_FIREBALL:
224 me->CastSpell(me->GetVictim(), SPELL_FIREBALL, false);
225 events.ScheduleEvent(EVENT_MILL_BASE_SPELL, 4000);
226 break;
227 case SPELL_ARCANE_MISSILES:
228 me->CastSpell(me->GetVictim(), SPELL_ARCANE_MISSILES, false);
229 events.ScheduleEvent(EVENT_MILL_BASE_SPELL, 9000);
230 break;
231 case SPELL_FROSTBOLT:
232 me->CastSpell(me->GetVictim(), SPELL_FROSTBOLT, false);
233 events.ScheduleEvent(EVENT_MILL_BASE_SPELL, 4000);
234 break;
235 default:
236 break;
237 }
238 break;
239 }
240
241 DoMeleeAttackIfReady();
242 }
243 };
