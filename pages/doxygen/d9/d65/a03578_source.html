196 void SetData(uint32 type, uint32 data) override
197 {
198 if (type < MAX_ENCOUNTER)
199 Encounter[type] = data;
200
201 if (data == DONE)
202 {
203 isMaidenOfGriefDead = type == BOSS_MAIDEN_OF_GRIEF || isMaidenOfGriefDead;
204 isKrystalusDead = type == BOSS_KRYSTALLUS || isKrystalusDead;
205 }
206
207 if (isMaidenOfGriefDead && isKrystalusDead)
208 if (GameObject* tribunalDoor = instance->GetGameObject(goTribunalDoorGUID))
209 tribunalDoor->SetGoState(GO_STATE_ACTIVE);
210
211 if (type == BOSS_TRIBUNAL_OF_AGES && data == DONE)
212 {
213 if (GameObject* pA = instance->GetGameObject(goAbedneumGUID))
214 pA->SetGoState(GO_STATE_ACTIVE);
215 if (GameObject* pF = instance->GetGameObject(goSkyRoomFloorGUID))
216 pF->SetGoState(GO_STATE_ACTIVE);
217
218 // Make sjonnir attackable
219 if (Creature* cr = instance->GetCreature(SjonnirGUID))
220 cr->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
221 }
222 if (type == BOSS_TRIBUNAL_OF_AGES && data == NOT_STARTED)
223 {
224 if (GameObject* pA = instance->GetGameObject(goAbedneumGUID))
225 pA->SetGoState(GO_STATE_READY);
226 if (GameObject* pF = instance->GetGameObject(goSkyRoomFloorGUID))
227 pF->SetGoState(GO_STATE_READY);
228 }
229
230 if (type == DATA_BRANN_ACHIEVEMENT)
231 {
232 brannAchievement = (bool)data;
233 return;
234 }
235 if (type == DATA_SJONNIR_ACHIEVEMENT)
236 {
237 sjonnirAchievement = (bool)data;
238 return;
239 }
240
241 if (data == DONE)
242 SaveToDB();
243 }

256 void Load(const char* strIn) override
257 {
258 if (!strIn)
259 {
260 OUT_LOAD_INST_DATA_FAIL;
261 return;
262 }
263
264 OUT_LOAD_INST_DATA(strIn);
265
266 char dataHead1, dataHead2, dataHead3;
267
268 std::istringstream loadStream(strIn);
269 loadStream >> dataHead1 >> dataHead2 >> dataHead3;
270
271 if (dataHead1 == 'H' && dataHead2 == 'O' && dataHead3 == 'S')
272 {
273 for (uint8 i = 0; i < MAX_ENCOUNTER; ++i)
274 {
275 loadStream >> Encounter[i];
276 if( Encounter[i] == IN_PROGRESS )
277 Encounter[i] = NOT_STARTED;
278 }
279 }
280 OUT_LOAD_INST_DATA_COMPLETE;
281 }
