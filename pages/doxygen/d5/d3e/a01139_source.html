35void FormationMgr::AddCreatureToGroup(uint32 groupId, Creature* member)
36{
37 Map* map = member->FindMap();
38 if (!map)
39 {
40 return;
41 }
42
43 CreatureGroupHolderType::iterator itr = map->CreatureGroupHolder.find(groupId);
44
45 //Add member to an existing group
46 if (itr != map->CreatureGroupHolder.end())
47 {
48 LOG_DEBUG("entities.unit", "Group found: {}, inserting creature {}, Group InstanceID {}", groupId, member->GetGUID().ToString(), member->GetInstanceId());
49 itr->second->AddMember(member);
50 }
51 //Create new group
52 else
53 {
54 LOG_DEBUG("entities.unit", "Group not found: {}. Creating new group.", groupId);
55 CreatureGroup* group = new CreatureGroup(groupId);
56 map->CreatureGroupHolder[groupId] = group;
57 group->AddMember(member);
58 }

80void FormationMgr::LoadCreatureFormations()
81{
82 uint32 const oldMSTime = getMSTime();
83 CreatureGroupMap.clear();
84
85 //Get group data
86 QueryResult result = WorldDatabase.Query("SELECT leaderGUID, memberGUID, dist, angle, groupAI, point_1, point_2 FROM creature_formations ORDER BY leaderGUID");
87 if (!result)
88 {
89 LOG_WARN("server.loading", ">> Loaded 0 creatures in formations. DB table `creature_formations` is empty!");
90 LOG_INFO("server.loading", " ");
91 return;
92 }
93
94 uint32 count = 0;
95 do
96 {
97 Field const* fields = result->Fetch();
98
99 //Load group member data
100 FormationInfo group_member;
101 group_member.leaderGUID = fields[0].Get<uint32>();
102 ObjectGuid::LowType const memberGUID = fields[1].Get<uint32>();
103 float const follow_dist = fields[2].Get<float>();
104 float const follow_angle = fields[3].Get<float>() * (static_cast<float>(M_PI) / 180);
105 group_member.groupAI = fields[4].Get<uint16>();
106 group_member.point_1 = fields[5].Get<uint16>();
107 group_member.point_2 = fields[6].Get<uint16>();
108
109 //If creature is group leader we may skip loading of dist/angle
110 if (group_member.leaderGUID != memberGUID)
111 {
112 if (!group_member.HasGroupFlag(std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_SUPPORTED)))
113 {
114 LOG_ERROR("sql.sql", "creature_formations table leader guid {} and member guid {} has unsupported GroupAI flag value ({}). Skipped", group_member.leaderGUID, memberGUID, group_member.groupAI);
115 continue;
116 }
117
118 if (!group_member.HasGroupFlag(std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_FOLLOW_LEADER)) && (follow_dist > 0.0f || follow_angle > 0.0f))
119 {
120 LOG_ERROR("sql.sql", "creature_formations table member guid {} and leader guid {} cannot have follow distance or follow angle because don't have GROUP_AI_FLAG_FOLLOW_LEADER flag. Values are not gonna be used", memberGUID, group_member.leaderGUID);
121 group_member.follow_dist = 0.0f;
122 group_member.follow_angle = 0.0f;
123 }
124 else
125 {
126 group_member.follow_dist = follow_dist;
127 group_member.follow_angle = follow_angle;
128 }
129 }
130 else
131 {
132 // Leader can have 0 AI flags - its allowed
133 if (group_member.groupAI && !group_member.HasGroupFlag(std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_SUPPORTED)))
134 {
135 LOG_ERROR("sql.sql", "creature_formations table leader guid {} and member guid {} has unsupported GroupAI flag value ({}). Skipped", group_member.leaderGUID, memberGUID, group_member.groupAI);
136 continue;
137 }
138
139 group_member.follow_dist = 0.0f;
140 group_member.follow_angle = 0.0f;
141 if (follow_dist > 0.0f || follow_angle > 0.0f)
142 {
143 LOG_ERROR("sql.sql", "creature_formations table member guid {} and leader guid {} cannot have follow distance or follow angle. Values are not gonna be used", memberGUID, group_member.leaderGUID);
144 }
145 }
146
147 if (!sObjectMgr->GetCreatureData(group_member.leaderGUID))
148 {
149 LOG_ERROR("sql.sql", "creature_formations table leader guid {} incorrect (does not exist). Skipped", group_member.leaderGUID);
150 continue;
151 }
152
153 if (!sObjectMgr->GetCreatureData(memberGUID))
154 {
155 LOG_ERROR("sql.sql", "creature_formations table member guid {} incorrect (does not exist). Skipped", memberGUID);
156 continue;
157 }
158
159 CreatureGroupMap[memberGUID] = group_member;
160 ++count;
161 } while (result->NextRow());
162
163 LOG_INFO("server.loading", ">> Loaded {} Creatures In Formations in {} ms", count, GetMSTimeDiffToNow(oldMSTime));
164 LOG_INFO("server.loading", " ");
165}

193void CreatureGroup::MemberEngagingTarget(Creature* member, Unit* target)
194{
195 uint8 const groupAI = sFormationMgr->CreatureGroupMap[member->GetSpawnId()].groupAI;
196 if (member == m_leader)
197 {
198 if (!(groupAI & std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_MEMBER_ASSIST_LEADER)))
199 {
200 return;
201 }
202 }
203 else if (!(groupAI & std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_LEADER_ASSIST_MEMBER)))
204 {
205 return;
206 }
207
208 for (auto const& itr : m_members)
209 {
210 Creature* pMember = itr.first;
211 if (m_leader) // avoid crash if leader was killed and reset.
212 LOG_DEBUG("entities.unit", "GROUP ATTACK: group instance id {} calls member instid {}", m_leader->GetInstanceId(), member->GetInstanceId());
213
214 //Skip one check
215 if (pMember == member)
216 continue;
217
218 if (!pMember->IsAlive())
219 continue;
220
221 if (pMember->GetVictim())
222 continue;
223
224 if (pMember->IsValidAttackTarget(target) && pMember->AI())
225 pMember->AI()->AttackStart(target);
226 }

229void CreatureGroup::MemberEvaded(Creature* member)
230{
231 uint8 const groupAI = sFormationMgr->CreatureGroupMap[member->GetSpawnId()].groupAI;
232 if (!(groupAI & std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_EVADE_MASK)))
233 {
234 return;
235 }
236
237 for (auto const& itr : m_members)
238 {
239 Creature* pMember = itr.first;
240 // This should never happen
241 if (!pMember)
242 {
243 continue;
244 }
245
246 if (pMember == member || pMember->IsInEvadeMode() || !itr.second.HasGroupFlag(std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_EVADE_MASK)))
247 {
248 continue;
249 }
250
251 if (itr.second.HasGroupFlag(std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_EVADE_TOGETHER)))
252 {
253 if (!pMember->IsAlive() || !pMember->IsInCombat())
254 {
255 continue;
256 }
257
258 if (pMember->IsAIEnabled)
259 {
260 if (CreatureAI* pMemberAI = pMember->AI())
261 {
262 pMemberAI->EnterEvadeMode();
263 }
264 }
265 }
266 else
267 {
268 if (pMember->IsAlive())
269 {
270 continue;
271 }
272
273 if (itr.second.HasGroupFlag(std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_DONT_RESPAWN_LEADER_ON_EVADE)) && pMember == m_leader)
274 {
275 continue;
276 }
277
278 pMember->Respawn();
279 }
280 }

283void CreatureGroup::FormationReset(bool dismiss, bool initMotionMaster)
284{
285 if (m_members.size() && !(m_members.begin()->second.HasGroupFlag(std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_FOLLOW_LEADER))))
286 {
287 return;
288 }
289
290 for (auto const& itr : m_members)
291 {
292 Creature* member = itr.first;
293 if (member && member != m_leader && member->IsAlive())
294 {
295 if (initMotionMaster)
296 {
297 if (dismiss)
298 {
299 member->GetMotionMaster()->Initialize();
300 }
301 else
302 {
303 member->GetMotionMaster()->MoveIdle();
304 }
305 LOG_DEBUG("entities.unit", "Set {} movement for member {}", dismiss ? "default" : "idle", member->GetGUID().ToString());
306 }
307 }
308 }
309 m_Formed = !dismiss;

312void CreatureGroup::LeaderMoveTo(float x, float y, float z, bool run)
313{
316 if (!m_leader)
317 {
318 return;
319 }
320
321 float pathDist = m_leader->GetExactDist(x, y, z);
322 float pathAngle = std::atan2(m_leader->GetPositionY() - y, m_leader->GetPositionX() - x);
323
324 for (auto const& itr : m_members)
325 {
326 Creature* member = itr.first;
327 FormationInfo const& pFormationInfo = itr.second;
328 if (member == m_leader || !member->IsAlive() || member->GetVictim() || !pFormationInfo.HasGroupFlag(std::underlying_type_t<GroupAIFlags>(GroupAIFlags::GROUP_AI_FLAG_FOLLOW_LEADER)))
329 {
330 continue;
331 }
332
333 // Xinef: If member is stunned / rooted etc don't allow to move him
334 if (member->HasUnitState(UNIT_STATE_NOT_MOVE))
335 {
336 continue;
337 }
338
339 // Xinef: this should be automatized, if turn angle is greater than PI/2 (90) we should swap formation angle
340 float followAngle = pFormationInfo.follow_angle;
341 if (static_cast<float>(M_PI) - std::fabs(std::fabs(m_leader->GetOrientation() - pathAngle) - static_cast<float>(M_PI)) > static_cast<float>(M_PI)* 0.5f)
342 {
343 // pussywizard: in both cases should be 2*M_PI - follow_angle
344 // pussywizard: also, GetCurrentWaypointID() returns 0..n-1, while point_1 must be > 0, so +1
345 // pussywizard: db table waypoint_data shouldn't have point id 0 and shouldn't have any gaps for this to work!
346 // if (m_leader->GetCurrentWaypointID()+1 == pFormationInfo->point_1 || m_leader->GetCurrentWaypointID()+1 == itr->second->point_2)
347 followAngle = Position::NormalizeOrientation(pFormationInfo.follow_angle + static_cast<float>(M_PI)); //(2 * M_PI) - itr->second->follow_angle;
348 }
349
350 float const followDist = pFormationInfo.follow_dist;
351
352 float dx = x + std::cos(followAngle + pathAngle) * followDist;
353 float dy = y + std::sin(followAngle + pathAngle) * followDist;
354 float dz = z;
355
356 Acore::NormalizeMapCoord(dx);
357 Acore::NormalizeMapCoord(dy);
358 member->UpdateGroundPositionZ(dx, dy, dz);
359
360 member->SetUnitMovementFlags(m_leader->GetUnitMovementFlags());
361 // pussywizard: setting the same movementflags is not enough, spline decides whether leader walks/runs, so spline param is now passed as "run" parameter to this function
362 if (run && member->IsWalking())
363 {
364 member->RemoveUnitMovementFlag(MOVEMENTFLAG_WALKING);
365 }
366 else if (!run && !member->IsWalking())
367 {
368 member->AddUnitMovementFlag(MOVEMENTFLAG_WALKING);
369 }
370
371 // xinef: if we move members to position without taking care of sizes, we should compare distance without sizes
372 // xinef: change members speed basing on distance - if too far speed up, if too close slow down
373 UnitMoveType const mtype = Movement::SelectSpeedType(member->GetUnitMovementFlags());
374 float const speedRate = m_leader->GetSpeedRate(mtype) * member->GetExactDist(dx, dy, dz) / pathDist;
375
376 if (speedRate > 0.01f) // don't move if speed rate is too low
377 {
378 member->SetSpeedRate(mtype, speedRate);
379 member->GetMotionMaster()->MovePoint(0, dx, dy, dz);
380 member->SetHomePosition(dx, dy, dz, pathAngle);
381 }
382 }
