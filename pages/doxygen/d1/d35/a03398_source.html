88 boss_festergutAI(Creature* creature) : BossAI(creature, DATA_FESTERGUT)

93 ObjectGuid _gasDummyGUID;
94 uint32 _maxInoculatedStack;
95 uint32 _inhaleCounter;

97 void Reset() override
98 {
99 _maxInoculatedStack = 0;
100 _inhaleCounter = 0;
101 _Reset();
102 events.Reset();
103 if (Creature* gasDummy = me->FindNearestCreature(NPC_GAS_DUMMY, 100.0f, true))
104 {
105 _gasDummyGUID = gasDummy->GetGUID();
106 for (uint8 i = 0; i < 3; ++i)
107 gasDummy->RemoveAurasDueToSpell(gaseousBlightVisual[i]);
108 }
109 }
110
111 void EnterCombat(Unit* who) override
112 {
113 if (!instance->CheckRequiredBosses(DATA_FESTERGUT, who->ToPlayer()))
114 {
115 EnterEvadeMode(EVADE_REASON_OTHER);
116 instance->DoCastSpellOnPlayers(LIGHT_S_HAMMER_TELEPORT);
117 return;
118 }
119
120 events.ScheduleEvent(EVENT_BERSERK, 300000);
121 events.ScheduleEvent(EVENT_INHALE_BLIGHT, urand(25000, 30000));
122 events.ScheduleEvent(EVENT_GAS_SPORE, urand(20000, 25000));
123 events.ScheduleEvent(EVENT_VILE_GAS, urand(30000, 40000), 1);
124 events.ScheduleEvent(EVENT_GASTRIC_BLOAT, urand(12500, 15000));
125 if (IsHeroic())
126 events.ScheduleEvent(EVENT_FESTERGUT_GOO, urand(15000, 20000));
127
128 me->setActive(true);
129 Talk(SAY_AGGRO);
130 DoZoneInCombat();
131
132 if (Creature* gasDummy = me->FindNearestCreature(NPC_GAS_DUMMY, 100.0f, true))
133 _gasDummyGUID = gasDummy->GetGUID();
134 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
135 professor->AI()->DoAction(ACTION_FESTERGUT_COMBAT);
136 }
137
138 void JustDied(Unit* /*killer*/) override
139 {
140 _JustDied();
141 Talk(SAY_DEATH);
142 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
143 professor->AI()->DoAction(ACTION_FESTERGUT_DEATH);
144
145 RemoveBlight();
146 }

148 void JustReachedHome() override

150 _JustReachedHome();
151 instance->SetBossState(DATA_FESTERGUT, FAIL);
152 }

167 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
168 {
169 if (spell->Id == PUNGENT_BLIGHT_HELPER)
170 target->RemoveAurasDueToSpell(INOCULATED_HELPER);
171 else if (Player* p = target->ToPlayer())
172 {
173 // Gaseous Blight damage
174 if (((spell->Id == 69159 || spell->Id == 70136 || spell->Id == 69161 || spell->Id == 70139 || spell->Id == 69163 || spell->Id == 70469) && p->GetQuestStatus(QUEST_RESIDUE_RENDEZVOUS_10) == QUEST_STATUS_INCOMPLETE) ||
175 ((spell->Id == 70135 || spell->Id == 70138 || spell->Id == 70468 || spell->Id == 70137 || spell->Id == 70140 || spell->Id == 70470) && p->GetQuestStatus(QUEST_RESIDUE_RENDEZVOUS_25) == QUEST_STATUS_INCOMPLETE))
176 p->CastSpell(p, SPELL_ORANGE_BLIGHT_RESIDUE, true);
177 }
178 }
179
180 void RemoveBlight()
181 {
182 if (Creature* gasDummy = ObjectAccessor::GetCreature(*me, _gasDummyGUID))
183 for (uint8 i = 0; i < 3; ++i)
184 {
185 me->RemoveAurasDueToSpell(gaseousBlight[i]);
186 gasDummy->RemoveAurasDueToSpell(gaseousBlightVisual[i]);
187 }
188 }
189
190 void UpdateAI(uint32 diff) override
191 {
192 if (!UpdateVictim())
193 return;
194
195 events.Update(diff);
196
197 if (me->HasUnitState(UNIT_STATE_CASTING))
198 return;
199
200 switch (events.ExecuteEvent())
201 {
202 case EVENT_BERSERK:
203 me->CastSpell(me, SPELL_BERSERK2, true);
204 Talk(SAY_BERSERK);
205 break;
206 case EVENT_INHALE_BLIGHT:
207 RemoveBlight();
208 if (_inhaleCounter == 3)
209 {
210 Talk(EMOTE_WARN_PUNGENT_BLIGHT);
211 Talk(SAY_PUNGENT_BLIGHT);
212 me->CastSpell(me, SPELL_PUNGENT_BLIGHT, false);
213 _inhaleCounter = 0;
214 events.RescheduleEvent(EVENT_GAS_SPORE, urand(20000, 25000));
215 }
216 else
217 {
218 me->CastSpell(me, SPELL_INHALE_BLIGHT, false);
219 // just cast and dont bother with target, conditions will handle it
220 ++_inhaleCounter;
221 if (_inhaleCounter < 3)
222 me->CastSpell(me, gaseousBlight[_inhaleCounter], true, nullptr, nullptr, me->GetGUID());
223 }
224
225 events.ScheduleEvent(EVENT_INHALE_BLIGHT, 34000);
226 break;
227 case EVENT_GAS_SPORE:
228 Talk(EMOTE_WARN_GAS_SPORE);
229 Talk(EMOTE_GAS_SPORE);
230 me->CastCustomSpell(SPELL_GAS_SPORE, SPELLVALUE_MAX_TARGETS, RAID_MODE<int32>(2, 3, 2, 3), me);
231 events.ScheduleEvent(EVENT_GAS_SPORE, urand(40000, 45000));
232 events.DelayEventsToMax(20000, 1); // delay EVENT_VILE_GAS
233 break;
234 case EVENT_VILE_GAS:
235 {
236 std::list<Unit*> targets;
237 uint32 minTargets = RAID_MODE<uint32>(3, 8, 3, 8);
238 SelectTargetList(targets, minTargets, SelectTargetMethod::Random, -5.0f, true);
239 float minDist = 0.0f;
240 if (targets.size() >= minTargets)
241 minDist = -5.0f;
242
243 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, minDist, true))
244 me->CastSpell(target, SPELL_VILE_GAS, false);
245 events.ScheduleEvent(EVENT_VILE_GAS, urand(28000, 35000), 1);
246 break;
247 }
248 case EVENT_GASTRIC_BLOAT:
249 me->CastSpell(me->GetVictim(), SPELL_GASTRIC_BLOAT, false);
250 events.ScheduleEvent(EVENT_GASTRIC_BLOAT, urand(15000, 17500));
251 break;
252 case EVENT_FESTERGUT_GOO:
253 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, NonTankTargetSelector(me)))
254 if (Creature* professor = ObjectAccessor::GetCreature(*me, instance->GetGuidData(DATA_PROFESSOR_PUTRICIDE)))
255 professor->CastSpell(target, SPELL_MALLABLE_GOO_H, true);
256 events.ScheduleEvent(EVENT_FESTERGUT_GOO, urand(15000, 20000));
257 default:
258 break;
259 }
260
261 DoMeleeAttackIfReady();
262 }
263
264 void SetData(uint32 type, uint32 data) override
265 {
266 if (type == DATA_INOCULATED_STACK && data > _maxInoculatedStack)
267 _maxInoculatedStack = data;
268 }
269
270 uint32 GetData(uint32 type) const override
271 {
272 if (type == DATA_INOCULATED_STACK)
273 return _maxInoculatedStack;
274
275 return 0;
276 }
277 };
