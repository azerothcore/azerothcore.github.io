88 void Reset() override

101 void EnterCombat(Unit* /*who*/) override
102 {
103 if (Creature* k = GetKrick())
104 k->AI()->Talk(SAY_AGGRO);
105 DoZoneInCombat();
106 events.Reset();
107 events.RescheduleEvent(EVENT_SPELL_TOXIC_WASTE, urand(3000, 5000));
108 events.RescheduleEvent(EVENT_SPELL_MIGHTY_KICK, urand(10000, 20000));
109 events.RescheduleEvent(EVENT_SPELL_SHADOW_BOLT, 10000);

113 pInstance->SetData(DATA_ICK, IN_PROGRESS);

116 void SpellHitTarget(Unit* target, SpellInfo const* spell) override
117 {
118 if (!target || !spell)
119 return;
120 if (spell->Id == SPELL_PURSUIT && target->GetTypeId() == TYPEID_PLAYER)
121 {
122 WorldPacket data;
123
124 ChatHandler::BuildChatPacket(data, CHAT_MSG_RAID_BOSS_EMOTE, LANG_UNIVERSAL, me, nullptr, EMOTE_ICK_CHASING);
125 target->ToPlayer()->GetSession()->SendPacket(&data);
126
127 AttackStart(target);
128 me->SetReactState(REACT_PASSIVE);
129 events.RescheduleEvent(EVENT_SET_REACT_AGGRESSIVE, 12000);

133 Creature* GetKrick()

141 void DamageTaken(Unit* /*doneBy*/, uint32& damage, DamageEffectType, SpellSchoolMask) override
142 {
143 if (damage >= me->GetHealth())
144 if (Creature* krick = GetKrick())
145 {
146 krick->InterruptNonMeleeSpells(true);
147 krick->RemoveAllAuras();
148 Position myPos(*me), exitPos;
149 float ang = me->GetOrientation() + 3 * M_PI / 2;
150 float dist = 3.0f;
151 exitPos.Relocate(myPos.GetPositionX() + dist * cos(ang), myPos.GetPositionY() + dist * std::sin(ang), 515.0f, M_PI);
152 exitPos.m_positionZ = me->GetMap()->GetHeight(exitPos.GetPositionX(), exitPos.GetPositionY(), exitPos.GetPositionZ());
153
154 if (exitPos.GetPositionZ() < 505.0f || exitPos.GetPositionZ() > 512.0f || !me->IsWithinLOS(exitPos.GetPositionX(), exitPos.GetPositionY(), exitPos.GetPositionZ()))
155 exitPos.Relocate(myPos);
156
157 krick->_ExitVehicle(&exitPos);
158 krick->AI()->DoAction(1);
159 }

162 void UpdateAI(uint32 diff) override
163 {
164 if (!UpdateVictim())
165 return;
166
167 if (me->GetVictim())
168 {
169 float x, y, z;
170 me->GetVictim()->GetPosition(x, y, z);
171 if (KrickCenterPos.GetExactDist(x, y, z) > 80.0f || z > KrickCenterPos.GetPositionZ() + 20.0f || z < KrickCenterPos.GetPositionZ() - 20.0f)
172 {
173 me->SetHealth(me->GetMaxHealth());
174 EnterEvadeMode();
175 return;
176 }
177 }
178
179 events.Update(diff);
180
181 if (me->HasUnitState(UNIT_STATE_CASTING | UNIT_STATE_STUNNED))
182 return;
183
184 switch(events.ExecuteEvent())
185 {
186 case 0:
187 break;
188 case EVENT_SPELL_TOXIC_WASTE:
189 if (Creature* k = GetKrick())
190 if (!k->HasUnitState(UNIT_STATE_CASTING))
191 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 40.0f, true))
192 {
193 k->CastSpell(target, SPELL_TOXIC_WASTE);
194 events.RepeatEvent(urand(7000, 10000));
195 break;
196 }
197 events.RepeatEvent(2500);
198 break;
199 case EVENT_SPELL_MIGHTY_KICK:
200 me->CastSpell(me->GetVictim(), SPELL_MIGHTY_KICK, false);
201 events.RepeatEvent(urand(20000, 25000));
202 break;
203 case EVENT_SPELL_SHADOW_BOLT:
204 if (Creature* k = GetKrick())
205 if (!k->HasUnitState(UNIT_STATE_CASTING))
206 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 35.0f, true))
207 {
208 k->CastSpell(target, SPELL_SHADOW_BOLT);
209 events.RepeatEvent(14000);
210 break;
211 }
212 events.RepeatEvent(2500);
213 break;
214 case EVENT_SET_REACT_AGGRESSIVE:
215 me->SetReactState(REACT_AGGRESSIVE);
216 if (!UpdateVictim())
217 return;
218
219 break;
220 case EVENT_SPECIAL:
221 switch(urand(0, 2))
222 {
223 case 0: // Pursuit
224 if (Creature* k = GetKrick())
225 k->AI()->Talk(RAND(SAY_TARGET_1, SAY_TARGET_2, SAY_TARGET_3));
226 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 70.0f, true))
227 me->CastSpell(target, SPELL_PURSUIT, false);
228 break;
229 case 1: // Poison Nova
230 if (Creature* k = GetKrick())
231 {
232 k->AI()->Talk(SAY_ORDER_BLOW);
233 Talk(EMOTE_ICK_POISON);
234 }
235 me->CastSpell(me, SPELL_POISON_NOVA, false);
236 break;
237 case 2: // Explosive Barrage
238 if (Creature* k = GetKrick())
239 {
240 k->AI()->Talk(SAY_ORDER_STOP);
241 k->AI()->Talk(EMOTE_KRICK_MINES);
242 k->InterruptNonMeleeSpells(false);
243 me->InterruptNonMeleeSpells(false);
244 k->CastSpell(k, SPELL_EXPLOSIVE_BARRAGE_KRICK, false);
245 me->CastSpell(me, SPELL_EXPLOSIVE_BARRAGE_ICK, false);
246 }
247 events.DelayEvents(20000);
248 break;
249 }
250 events.RepeatEvent(urand(25000, 30000));
251 break;
252 }
253
254 DoMeleeAttackIfReady();
255 }

271 if (who->GetTypeId() == TYPEID_PLAYER)
272 if (Creature* k = GetKrick())
273 k->AI()->Talk(RAND(SAY_SLAY_1, SAY_SLAY_2));
274 }
