265 reviveTimer += diff;
266 if (reviveTimer >= 12000)
267 {
268 for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
269 {
270 if (Creature* cr = ObjectAccessor::GetCreature(*me, (*itr)))
271 {

280 events.ScheduleEvent(EVENT_THADDIUS_INIT, 750);
281 }
282 return;
283 }
284
285 if (!UpdateVictim())
286 return;
287
288 events.Update(diff);
289 if (me->HasUnitState(UNIT_STATE_CASTING))
290 return;
291
292 if (summonTimer) // Revive
293 {
294 summonTimer += diff;
295 if (summonTimer >= 5000)
296 {
297 summons.DoAction(ACTION_RESTORE);
298 summonTimer = 0;
299 }
300 }
301
302 switch (events.ExecuteEvent())
303 {
304 case EVENT_THADDIUS_INIT:
305 {
306 me->RemoveAllAuras();
307 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
308 for (SummonList::const_iterator itr = summons.begin(); itr != summons.end(); ++itr)
309 {
310 if (Creature* cr = ObjectAccessor::GetCreature(*me, (*itr)))

312 if (cr->GetEntry() == NPC_TESLA_COIL)
313 {
314 Unit::Kill(cr, cr);
315 }
316 }
317 }
318 if (GameObject* go = me->FindNearestGameObject(GO_TESLA_COIL_LEFT, 100.0f))
319 {
320 go->SetGoState(GO_STATE_READY);
321 }
322 if (GameObject* go = me->FindNearestGameObject(GO_TESLA_COIL_RIGHT, 100.0f))
323 {
324 go->SetGoState(GO_STATE_READY);
325 }
326 me->CastSpell(me, SPELL_THADDIUS_VISUAL_LIGHTNING, true);
327 events.ScheduleEvent(EVENT_THADDIUS_ENTER_COMBAT, 1000);
328 break;
329 }
330 case EVENT_THADDIUS_ENTER_COMBAT:
331 Talk(SAY_AGGRO);
332 me->SetReactState(REACT_AGGRESSIVE);
333 me->SetControlled(false, UNIT_STATE_STUNNED);
334 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
335 events.ScheduleEvent(EVENT_THADDIUS_CHAIN_LIGHTNING, 14000);
336 events.ScheduleEvent(EVENT_THADDIUS_BERSERK, 360000);
337 events.ScheduleEvent(EVENT_THADDIUS_POLARITY_SHIFT, 30000);
338 events.ScheduleEvent(EVENT_ALLOW_BALL_LIGHTNING, 5000);
339 return;
340 case EVENT_THADDIUS_BERSERK:
341 me->CastSpell(me, SPELL_BERSERK, true);
342 break;
343 case EVENT_THADDIUS_CHAIN_LIGHTNING:
344 me->CastSpell(me->GetVictim(), RAID_MODE(SPELL_CHAIN_LIGHTNING_10, SPELL_CHAIN_LIGHTNING_25), false);
345 events.RepeatEvent(15000);
346 break;
347 case EVENT_THADDIUS_POLARITY_SHIFT:
348 me->CastSpell(me, SPELL_POLARITY_SHIFT, false);
349 events.RepeatEvent(30000);
350 break;
351 case EVENT_ALLOW_BALL_LIGHTNING:
352 ballLightningEnabled = true;
353 break;
354 }
355
356 if (me->IsWithinMeleeRange(me->GetVictim()))
357 {
358 DoMeleeAttackIfReady();
359 }
360 else if (ballLightningEnabled)
361 {
362 if (Unit* target = SelectTarget(SelectTargetMethod::MaxThreat))
363 {
364 me->CastSpell(target, SPELL_BALL_LIGHTNING, false);
365 }
366 }

612 class spell_thaddius_pos_neg_charge_SpellScript : public SpellScript
613 {
614 PrepareSpellScript(spell_thaddius_pos_neg_charge_SpellScript);
615
616 void HandleTargets(std::list<WorldObject*>& targets)
617 {
618 uint8 count = 0;
619 for (auto& ihit : targets)
620 {
621 if (ihit->GetGUID() != GetCaster()->GetGUID())
622 {
623 if (Player* target = ihit->ToPlayer())
624 {
625 if (target->HasAura(GetTriggeringSpell()->Id))
626 {
627 ++count;
628 }
629 }
630 }
631 }
632
633 if (count)
634 {
635 uint32 spellId = GetSpellInfo()->Id == SPELL_POSITIVE_CHARGE ? SPELL_POSITIVE_CHARGE_STACK : SPELL_NEGATIVE_CHARGE_STACK;
636 GetCaster()->SetAuraStack(spellId, GetCaster(), count);
637 }
638 }
639
640 void HandleDamage(SpellEffIndex /*effIndex*/)
641 {
642 if (!GetTriggeringSpell())
643 return;
644
645 Unit* target = GetHitUnit();
646 if (!target)
647 return;
648
649 if (target->HasAura(GetTriggeringSpell()->Id) || target->GetTypeId() != TYPEID_PLAYER)
650 {
651 SetHitDamage(0);
652 }
653 else if (target->GetInstanceScript())
654 {
655 target->GetInstanceScript()->SetData(DATA_CHARGES_CROSSED, 0);
656 }
657 }
658
659 void Register() override
660 {
661 OnEffectHitTarget += SpellEffectFn(spell_thaddius_pos_neg_charge_SpellScript::HandleDamage, EFFECT_0, SPELL_EFFECT_SCHOOL_DAMAGE);
662 OnObjectAreaTargetSelect += SpellObjectAreaTargetSelectFn(spell_thaddius_pos_neg_charge_SpellScript::HandleTargets, EFFECT_0, TARGET_UNIT_SRC_AREA_ALLY);
663 }
664 };

667 {
