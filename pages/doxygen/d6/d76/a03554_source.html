91 struct boss_volkhanAI : public ScriptedAI
92 {
93 boss_volkhanAI(Creature* creature) : ScriptedAI(creature), summons(creature)
94 {
95 m_pInstance = creature->GetInstanceScript();
96 }
97
98 InstanceScript* m_pInstance;
99 EventMap events;
100 SummonList summons;
101 uint8 HealthCheck;
102 float x, y, z;
103 uint8 PointID;
104 uint8 ShatteredCount;
105
106 void Reset() override
107 {
108 x = y = z = PointID = ShatteredCount = 0;
109 HealthCheck = 100;
110 events.Reset();
111 summons.DespawnAll();
112 me->SetSpeed(MOVE_RUN, 1.2f, true);
113 me->SetReactState(REACT_AGGRESSIVE);
114
115 if (m_pInstance)
116 {
117 m_pInstance->SetData(TYPE_VOLKHAN, NOT_STARTED);
118 m_pInstance->SetData(DATA_VOLKHAN_ACHIEVEMENT, true);
119 }
120 }
121
122 void EnterCombat(Unit*) override
123 {
124 me->SetInCombatWithZone();
125 Talk(SAY_AGGRO);
126
127 if (m_pInstance)
128 m_pInstance->SetData(TYPE_VOLKHAN, IN_PROGRESS);
129
130 ScheduleEvents(false);
131 }
132
133 void JustDied(Unit*) override
134 {
135 Talk(SAY_DEATH);
136
137 summons.DespawnAll();
138
139 if (m_pInstance)
140 m_pInstance->SetData(TYPE_VOLKHAN, DONE);
141 }
142
143 void GetNextPos()
144 {
145 if (me->GetPositionY() < -180)
146 {
147 if (me->GetPositionX() > 1330)
148 x = 1355;
149 else
150 x = 1308;
151
152 y = -178;
153 z = 52.5f;
154 }
155 else if (me->GetPositionY() < -145)
156 {
157 if (me->GetPositionX() > 1330)
158 x = 1355;
159 else
160 x = 1308;
161
162 y = -137;
163 z = 52.5f;
164 }
165 else if (me->GetPositionY() < -130)
166 {
167 if (me->GetPositionX() > 1330)
168 x = 1343;
169 else
170 x = 1320;
171
172 y = -123;
173 z = 56.7f;
174 }
175 else
176 {
177 PointID = POINT_ANVIL;
178 x = 1327;
179 y = -96;
180 z = 56.7f;
181 }
182 }
183
184 void KilledUnit(Unit* victim) override
185 {
186 if (victim->GetTypeId() != TYPEID_PLAYER)
187 return;
188
189 Talk(SAY_SLAY);
190 }
191
192 void ScheduleEvents(bool anvil)
193 {
194 events.SetPhase(1);
195 events.RescheduleEvent(EVENT_HEAT, 8000, 0, 1);
196 events.RescheduleEvent(EVENT_SHATTER, 10000, 0, 1);
197 events.RescheduleEvent(EVENT_CHECK_HEALTH, anvil ? 1000 : 6000, 0, 1);
198 events.RescheduleEvent(EVENT_POSITION, 4000, 0, 1);
199 }
200
201 void JustSummoned(Creature* summon) override
202 {
203 summons.Summon(summon);
204 if (summon->GetEntry() == NPC_MOLTEN_GOLEM)
205 {
206 summon->SetFaction(me->GetFaction());
207
208 if (Unit* target = SelectTarget(SelectTargetMethod::Random))
209 summon->AI()->AttackStart(target);
210 }
211 }
212
213 void DoAction(int32 param) override
214 {
215 if (param == ACTION_DESTROYED)
216 {
217 ShatteredCount++;
218 if (ShatteredCount > 4)
219 m_pInstance->SetData(DATA_VOLKHAN_ACHIEVEMENT, false);
220 }
221 }
222
223 void MovementInform(uint32 type, uint32 id) override
224 {
225 if (type != POINT_MOTION_TYPE)
226 return;
227
228 if (id == POINT_ANVIL)
229 {
230 me->SetSpeed(MOVE_RUN, 1.2f, true);
231 me->SetReactState(REACT_AGGRESSIVE);
232 me->CastSpell(me, SPELL_TEMPER, false);
233 PointID = 0;
234 ScheduleEvents(true);
235
236 // update orientation at server
237 me->SetOrientation(2.19f);
238
239 // and client
240 WorldPacket data;
241 me->BuildHeartBeatMsg(&data);
242 me->SendMessageToSet(&data, false);
243 me->SetControlled(true, UNIT_STATE_ROOT);
244 }
245 else
246 events.ScheduleEvent(EVENT_MOVE_TO_ANVIL, 0, 0, 2);
247 }
248
249 void SpellHitTarget(Unit* /*who*/, SpellInfo const* spellInfo) override
250 {
251 if (spellInfo->Id == SPELL_TEMPER)
252 {
253 me->CastSpell(me, SPELL_SUMMON_MOLTEN_GOLEM, true);
254 me->CastSpell(me, SPELL_SUMMON_MOLTEN_GOLEM, true);
255 me->GetMotionMaster()->MoveChase(me->GetVictim());
256 me->SetControlled(false, UNIT_STATE_ROOT);
257 }
258 }
259
260 void GoToAnvil()
261 {
262 events.SetPhase(2);
263 HealthCheck -= 20;
264 me->SetSpeed(MOVE_RUN, 4.0f, true);
265 me->SetReactState(REACT_PASSIVE);
266
267 Talk(SAY_FORGE);
268
269 if (me->GetMotionMaster()->GetCurrentMovementGeneratorType() == CHASE_MOTION_TYPE)
270 me->GetMotionMaster()->MovementExpired();
271
272 events.ScheduleEvent(EVENT_MOVE_TO_ANVIL, 0, 0, 2);
273 }
274
275 void UpdateAI(uint32 diff) override
276 {
277 //Return since we have no target
278 if (!UpdateVictim())
279 return;
280
281 events.Update(diff);
282
283 if (me->HasUnitState(UNIT_STATE_CASTING))
284 return;
285
286 switch (events.ExecuteEvent())
287 {
288 case EVENT_HEAT:
289 me->CastSpell(me, me->GetMap()->IsHeroic() ? SPELL_HEAT_H : SPELL_HEAT_N, true);
290 events.RepeatEvent(8000);
291 break;
292 case EVENT_CHECK_HEALTH:
293 if (HealthBelowPct(HealthCheck))
294 GoToAnvil();
295
296 events.RepeatEvent(1000);
297 return;
298 case EVENT_SHATTER:
299 {
300 events.RepeatEvent(10000);
301 summons.DoAction(ACTION_SHATTER);
302 break;
303 }
304 case EVENT_MOVE_TO_ANVIL:
305 GetNextPos();
306 me->GetMotionMaster()->MovePoint(PointID, x, y, z);
307 return;
308 case EVENT_POSITION:
309 if (me->GetDistance(1331.9f, -106, 56) > 95)
310 EnterEvadeMode();
311 else
312 events.RepeatEvent(4000);
313
314 return;
315 }
316
317 DoMeleeAttackIfReady();
318 }
319 };

339 EventMap events;
340 InstanceScript* m_pInstance;

342 void Reset() override
343 {
344 events.Reset();
345 events.ScheduleEvent(EVENT_BLAST, 7000);
346 events.ScheduleEvent(EVENT_IMMOLATION, 3000);

348
349 void DamageTaken(Unit*, uint32& uiDamage, DamageEffectType, SpellSchoolMask) override
350 {
351 if (me->GetEntry() == NPC_BRITTLE_GOLEM)
352 {
353 uiDamage = 0;
354 return;
355 }
356
357 if (uiDamage >= me->GetHealth())
358 {
359 me->UpdateEntry(NPC_BRITTLE_GOLEM, 0, false);
360 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE | UNIT_FLAG_NOT_SELECTABLE | UNIT_FLAG_DISABLE_MOVE);
361 me->SetHealth(me->GetMaxHealth());
362 me->RemoveAllAuras();
363 me->AttackStop();
364 uiDamage = 0;
365
366 if (me->IsNonMeleeSpellCast(false))
367 me->InterruptNonMeleeSpells(false);
368
369 me->SetControlled(true, UNIT_STATE_STUNNED);
370 }
371 }
372
373 void DoAction(int32 param) override
374 {
375 if (me->GetEntry() == NPC_BRITTLE_GOLEM && param == ACTION_SHATTER)
376 {
377 if (Creature* volkhan = ObjectAccessor::GetCreature(*me, m_pInstance->GetGuidData(TYPE_VOLKHAN)))
378 volkhan->AI()->DoAction(ACTION_DESTROYED);
379
380 me->CastSpell(me, me->GetMap()->IsHeroic() ? SPELL_SHATTER_H : SPELL_SHATTER_N, true);
381 me->DespawnOrUnsummon(500);
382 }
383 }
384
385 void UpdateAI(uint32 diff) override
386 {
387 //Return since we have no target or if we are frozen
388 if (!UpdateVictim() || me->GetEntry() == NPC_BRITTLE_GOLEM)
389 return;
390
391 events.Update(diff);
392
393 if (me->HasUnitState(UNIT_STATE_CASTING))
394 return;
395
396 switch (events.ExecuteEvent())
397 {
398 case EVENT_BLAST:
399 me->CastSpell(me, SPELL_BLAST_WAVE, false);
400 events.RepeatEvent(14000);
401 break;
402 case EVENT_IMMOLATION:
403 me->CastSpell(me->GetVictim(), me->GetMap()->IsHeroic() ? SPELL_IMMOLATION_STRIKE_H : SPELL_IMMOLATION_STRIKE_N, false);
404 events.RepeatEvent(5000);
405 break;
406 }
407
408 DoMeleeAttackIfReady();
409 }
410 };

474 void MoveInLineOfSight(Unit* who) override
475 {
476 if (_attackGUID)
477 ScriptedAI::MoveInLineOfSight(who);
478 else if (_isActive && who->GetTypeId() == TYPEID_PLAYER)

490 void EnterCombat(Unit*) override
491 {
492 events.Reset();
493 if (me->GetEntry() == 28961) // NPC_TITANIUM_SIEGEBREAKER

495 events.ScheduleEvent(EVENT_PIERCING_HOWL, 10000 + rand() % 15000);
496 events.ScheduleEvent(EVENT_PENETRATING_STRIKE, 5000 + rand() % 5000);
497 events.ScheduleEvent(EVENT_FRIGHTENING_SHOUT, 20000 + rand() % 8000);
498 events.ScheduleEvent(EVENT_BLADE_TURNING, 12000);
499 }
500 else
501 {
502 events.ScheduleEvent(EVENT_THROW, 10000 + rand() % 15000);
503 events.ScheduleEvent(EVENT_DEADLY_THROW, 15000 + rand() % 15000);
504 events.ScheduleEvent(EVENT_DEFLECTION, 15000);

507
508 void AttackStart(Unit* who) override
509 {
510 if (!_attackGUID || !_isActive)
511 return;
512 ScriptedAI::AttackStart(who);

514
515 void UpdateAI(uint32 diff) override
516 {
517 if (!_isActive && !_attackGUID)
518 return;
519
520 events.Update(diff);
521 uint32 eventId = events.ExecuteEvent();
522
523 if (eventId == EVENT_UNFREEZE)
524 {
525 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
526 me->CastSpell(me, SPELL_AWAKEN, true);
527 me->RemoveAllAuras();
528 _isActive = true;
529 if (Unit* target = ObjectAccessor::GetUnit(*me, _attackGUID))
530 AttackStart(target);
531 return;
532 }
533
534 //Return since we have no target or if we are disabled from fight
535 if (!UpdateVictim())
536 return;
537
538 if (me->HasUnitState(UNIT_STATE_CASTING))
539 return;
540
541 switch (eventId)
542 {
543 case EVENT_PIERCING_HOWL:
544 me->CastSpell(me->GetVictim(), SPELL_PIERCING_HOWL, false);
545 events.RepeatEvent(10000 + rand() % 1500);
546 break;
547 case EVENT_PENETRATING_STRIKE:
548 me->CastSpell(me->GetVictim(), SPELL_PENETRATING_STRIKE, false);
549 events.RepeatEvent(5000 + rand() % 5000);
550 break;

553 events.RepeatEvent(20000 + rand() % 8000);
554 break;
