195 void FireWall()
196 {
197 for (uint8 i = 0; i < 4; ++i)
198 {
199 uint8 WallNum = i == 0 || i == 2 ? 3 : 2;

201 for (uint8 j = 0; j < WallNum; j++)
202 {
203 Creature* wall = WallNum == 3
204 ? me->SummonCreature(NPC_FIRE_BOMB, FireWallCoords[i][0], FireWallCoords[i][1] + 5 * (j - 1), FireWallCoords[i][2], FireWallCoords[i][3], TEMPSUMMON_TIMED_DESPAWN, 15000)
205 : me->SummonCreature(NPC_FIRE_BOMB, FireWallCoords[i][0] - 2 + 4 * j, FireWallCoords[i][1], FireWallCoords[i][2], FireWallCoords[i][3], TEMPSUMMON_TIMED_DESPAWN, 15000);

271 void HandleBombSequence()
272 {
273 if (BombCount < 40)
274 {
275 if (Unit* FireBomb = ObjectAccessor::GetUnit(*me, FireBombGUIDs[BombCount]))
276 {
277 FireBomb->RemoveUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
278 DoCast(FireBomb, SPELL_FIRE_BOMB_THROW, true);
279 FireBomb->SetUnitFlag(UNIT_FLAG_NOT_SELECTABLE);
280 }
281 ++BombCount;
282 if (BombCount == 40)
283 {
284 BombSequenceTimer = 5000;
285 }
286 else BombSequenceTimer = 100;
287 }
288 else
289 {
290 Boom();
291 isBombing = false;
292 BombTimer = urand(20000, 40000);
293 me->RemoveAurasDueToSpell(SPELL_FIRE_BOMB_CHANNEL);
294 if (EnrageTimer <= 10000)
295 EnrageTimer = 0;
296 else
297 EnrageTimer -= 10000;
298 }

301 void UpdateAI(uint32 diff) override
302 {
303 if (isFlameBreathing)
304 {
305 if (!me->IsNonMeleeSpellCast(false))
306 isFlameBreathing = false;
307 else
308 return;
309 }
310
311 if (isBombing)
312 {
313 if (BombSequenceTimer <= diff)
314 HandleBombSequence();
315 else
316 BombSequenceTimer -= diff;
317 return;
318 }
319
320 if (!UpdateVictim())
321 return;
322
323 //enrage if under 25% hp before 5 min.
324 if (!enraged && HealthBelowPct(25))
325 EnrageTimer = 0;
326
327 if (EnrageTimer <= diff)
328 {
329 if (!enraged)
330 {
331 DoCast(me, SPELL_ENRAGE, true);
332 enraged = true;
333 EnrageTimer = 300000;
334 }
335 else
336 {
337 Talk(SAY_BERSERK);
338 DoCast(me, SPELL_BERSERK, true);
339 EnrageTimer = 300000;
340 }
341 }
342 else EnrageTimer -= diff;
343
344 if (BombTimer <= diff)
345 {
346 Talk(SAY_FIRE_BOMBS);
347
348 me->AttackStop();
349 me->GetMotionMaster()->Clear();
350 me->NearTeleportTo(JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2], me->GetOrientation());
351 me->StopMovingOnCurrentPos();
352 DoCast(me, SPELL_FIRE_BOMB_CHANNEL, false);
353 //DoTeleportPlayer(me, JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2], 0);
354 //DoCast(me, SPELL_TELE_TO_CENTER, true);
355
356 FireWall();
357 SpawnBombs();
358 isBombing = true;
359 BombSequenceTimer = 100;
360
361 //Teleport every Player into the middle
362 Map* map = me->GetMap();
363 if (!map->IsDungeon())
364 return;
365
366 Map::PlayerList const& PlayerList = map->GetPlayers();
367 for (Map::PlayerList::const_iterator i = PlayerList.begin(); i != PlayerList.end(); ++i)
368 if (Player* i_pl = i->GetSource())
369 if (i_pl->IsAlive())
370 DoTeleportPlayer(i_pl, JanalainPos[0][0] - 5 + rand() % 10, JanalainPos[0][1] - 5 + rand() % 10, JanalainPos[0][2], 0);
371 //DoCast(Temp, SPELL_SUMMON_PLAYERS, true) // core bug, spell does not work if too far
372 return;
373 }
374 else BombTimer -= diff;
375
376 if (!noeggs)
377 {
378 if (HealthBelowPct(35))
379 {
380 Talk(SAY_ALL_EGGS);
381
382 me->AttackStop();
383 me->GetMotionMaster()->Clear();
384 me->NearTeleportTo(JanalainPos[0][0], JanalainPos[0][1], JanalainPos[0][2], me->GetOrientation());
385 me->StopMovingOnCurrentPos();
386 DoCast(me, SPELL_HATCH_ALL, false);
387 HatchAllEggs(2);
388 noeggs = true;
389 }
390 else if (HatcherTimer <= diff)
391 {
392 if (HatchAllEggs(0))
393 {
394 Talk(SAY_SUMMON_HATCHER);
395 me->SummonCreature(NPC_AMANI_HATCHER, hatcherway[0][0][0], hatcherway[0][0][1], hatcherway[0][0][2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
396 me->SummonCreature(NPC_AMANI_HATCHER, hatcherway[1][0][0], hatcherway[1][0][1], hatcherway[1][0][2], 0, TEMPSUMMON_CORPSE_TIMED_DESPAWN, 10000);
397 HatcherTimer = 90000;
398 }
399 else
400 noeggs = true;
401 }
402 else HatcherTimer -= diff;
403 }
404
405 DoMeleeAttackIfReady();
406
407 if (FireBreathTimer <= diff)
408 {
409 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0))
410 {
411 me->AttackStop();
412 me->GetMotionMaster()->Clear();
413 DoCast(target, SPELL_FLAME_BREATH, false);
414 me->StopMoving();
415 isFlameBreathing = true;
416 }
417 FireBreathTimer = 8000;
418 }
419 else FireBreathTimer -= diff;
420 }
