106class TransportBase
107{
108protected:
109 TransportBase() = default;

112public:
114 virtual void CalculatePassengerPosition(float& x, float& y, float& z, float* o = nullptr) const = 0;
115
117 virtual void CalculatePassengerOffset(float& x, float& y, float& z, float* o = nullptr) const = 0;
118
119protected:
120 static void CalculatePassengerPosition(float& x, float& y, float& z, float* o, float transX, float transY, float transZ, float transO)
121 {
122 float inx = x, iny = y, inz = z;
123 if (o)
124 *o = Position::NormalizeOrientation(transO + *o);
125
126 x = transX + inx * std::cos(transO) - iny * std::sin(transO);
127 y = transY + iny * std::cos(transO) + inx * std::sin(transO);
128 z = transZ + inz;
129 }
130
131 static void CalculatePassengerOffset(float& x, float& y, float& z, float* o, float transX, float transY, float transZ, float transO)
132 {
133 if (o)
134 *o = Position::NormalizeOrientation(*o - transO);
135
136 z -= transZ;
137 y -= transY; // y = searchedY * std::cos(o) + searchedX * std::sin(o)
138 x -= transX; // x = searchedX * std::cos(o) + searchedY * std::sin(o + pi)
139 float inx = x, iny = y;
140 y = (iny - inx * std::tan(transO)) / (std::cos(transO) + std::sin(transO) * std::tan(transO));
141 x = (inx + iny * std::tan(transO)) / (std::cos(transO) + std::sin(transO) * std::tan(transO));
142 }
143};
