117 void Reset() override
118 {
119 events.Reset();
120 summons.DespawnAll();
121 Phase = 1;
122 me->SetDisplayId(me->GetNativeDisplayId());
123 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
124 me->SetImmuneToAll(true);
125 me->SetReactState(REACT_PASSIVE);

127 pInstance->SetData(BOSS_BLACK_KNIGHT, NOT_STARTED);

138 void DamageTaken(Unit*, uint32& damage, DamageEffectType, SpellSchoolMask) override
139 {
140 if (me->HasUnitFlag(UNIT_FLAG_NON_ATTACKABLE))
141 {
142 damage = 0;
143 return;
144 }
145
146 if( Phase < 3 && damage >= me->GetHealth() )
147 {
148 damage = 0;
149 me->SetHealth(me->GetMaxHealth());
150 events.Reset();
151 me->SetUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
152 me->RemoveAllAuras();
153 me->SetControlled(true, UNIT_STATE_STUNNED);
154 me->CastSpell(me, SPELL_BK_GHOUL_EXPLODE, true);
155 summons.clear();
156
157 me->CastSpell(me, SPELL_BK_FEIGN_DEATH, true);
158 me->SetUnitFlag(UNIT_FLAG_PREVENT_EMOTES_FROM_CHAT_TEXT);
159 me->SetUnitFlag2(UNIT_FLAG2_FEIGN_DEATH);
160 me->SetDynamicFlag(UNIT_DYNFLAG_DEAD);
161 me->AddUnitState(UNIT_STATE_DIED);
162 }
163 }
164
165 void DoAction(int32 param) override
166 {
167 if( param == -1 )
168 {
169 summons.DespawnAll();
170 }
171 else if( param == 1 )
172 {
173 if( !pInstance )
174 return;
175
176 pInstance->SetData(BOSS_BLACK_KNIGHT, IN_PROGRESS);
177 Talk(TEXT_BK_AGGRO);
178 me->CastSpell((Unit*)nullptr, (pInstance->GetData(DATA_TEAMID_IN_INSTANCE) == TEAM_HORDE ? SPELL_RAISE_DEAD_JAEREN : SPELL_RAISE_DEAD_ARELAS), false);
179 if( Creature* announcer = pInstance->instance->GetCreature(pInstance->GetGuidData(DATA_ANNOUNCER)) )
180 announcer->DespawnOrUnsummon();
181
182 events.Reset();
183 events.ScheduleEvent(EVENT_ANNOUNCER_SAY_ZOMBIE, 2500);
184 events.ScheduleEvent(EVENT_SPELL_PLAGUE_STRIKE, urand(7000, 9000));
185 events.ScheduleEvent(EVENT_SPELL_ICY_TOUCH, urand(3500, 7000));
186 events.ScheduleEvent(EVENT_SPELL_DEATH_RESPITE, urand(13000, 15000));
187 events.ScheduleEvent(EVENT_SPELL_OBLITERATE, urand(11000, 19000));
188 }
189 }
190
191 void SpellHitTarget(Unit* /*target*/, SpellInfo const* spell) override
192 {
193 switch( spell->Id )
194 {
195 case SPELL_BLACK_KNIGHT_RES:
196 me->SetHealth(me->GetMaxHealth());
197 me->RemoveUnitFlag(UNIT_FLAG_NON_ATTACKABLE);
198 me->SetImmuneToAll(false);
199 me->SetControlled(false, UNIT_STATE_STUNNED);
200
201 me->RemoveUnitFlag(UNIT_FLAG_PREVENT_EMOTES_FROM_CHAT_TEXT);
202 me->RemoveUnitFlag2(UNIT_FLAG2_FEIGN_DEATH);
203 me->RemoveDynamicFlag(UNIT_DYNFLAG_DEAD);
204 me->ClearUnitState(UNIT_STATE_DIED);
205
206 ++Phase;
207
208 switch( Phase )
209 {
210 case 2:
211 me->SetDisplayId(MODEL_SKELETON);
212 Talk(TEXT_BK_SKELETON_RES);
213 me->CastSpell(me, SPELL_ARMY_DEAD, false);
214
215 events.Reset();
216 events.ScheduleEvent(EVENT_SPELL_PLAGUE_STRIKE, urand(7000, 9000));
217 events.ScheduleEvent(EVENT_SPELL_ICY_TOUCH, urand(3500, 7000));
218 events.ScheduleEvent(EVENT_SPELL_OBLITERATE, urand(11000, 19000));
219 events.ScheduleEvent(EVENT_SPELL_DESECRATION, urand(2000, 3000));
220 break;
221 case 3:
222 me->SetDisplayId(MODEL_GHOST);
223 Talk(TEXT_BK_GHOST_RES);
224
225 events.Reset();
226 events.ScheduleEvent(EVENT_SPELL_DEATH_BITE, 2000);
227 events.ScheduleEvent(EVENT_SPELL_MARKED_DEATH, 1000);
228 break;
229 default:
230 EnterEvadeMode(EVADE_REASON_OTHER);
231 break;
232 }
233 break;
234 }
235 }

237 void UpdateAI(uint32 diff) override
238 {
239 if( !UpdateVictim() )
240 return;
241
242 events.Update(diff);
243
244 if( me->HasUnitState(UNIT_STATE_CASTING) )
245 return;
246
247 switch( events.ExecuteEvent() )
248 {
249 case 0:
250 break;
251 case EVENT_ANNOUNCER_SAY_ZOMBIE:
252 if( pInstance && !summons.empty() )
253 if( Creature* ghoul = pInstance->instance->GetCreature(*summons.begin()) )
254 ghoul->Yell("[Zombie] .... . Brains ....", LANG_UNIVERSAL);
255
256 break;
257 case EVENT_SPELL_PLAGUE_STRIKE:
258 if( me->GetVictim() )
259 me->CastSpell(me->GetVictim(), SPELL_PLAGUE_STRIKE, false);
260 events.RepeatEvent(urand(10000, 12000));
261 break;
262 case EVENT_SPELL_ICY_TOUCH:
263 if( me->GetVictim() )
264 me->CastSpell(me->GetVictim(), SPELL_ICY_TOUCH, false);
265 events.RepeatEvent(urand(5000, 6000));
266 break;
267 case EVENT_SPELL_DEATH_RESPITE:
268 if( Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true) )
269 me->CastSpell(target, SPELL_DEATH_RESPITE, false);
270 events.RepeatEvent(urand(13000, 15000));
271 break;
272 case EVENT_SPELL_OBLITERATE:
273 if( me->GetVictim() )
274 me->CastSpell(me->GetVictim(), SPELL_OBLITERATE, false);
275 events.RepeatEvent(urand(15000, 17000));
276 break;
277 case EVENT_SPELL_DESECRATION:
278 if( Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 50.0f, true) )
279 me->CastSpell(target, SPELL_DESECRATION, false);
280 events.RepeatEvent(urand(14000, 17000));
281 break;
282 case EVENT_SPELL_DEATH_BITE:
283 me->CastSpell((Unit*)nullptr, SPELL_DEATH_BITE, false);
284 events.RepeatEvent(urand(2000, 4000));
285 break;
286 case EVENT_SPELL_MARKED_DEATH:
287 if( Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 0.000000f, true) )
288 me->CastSpell(target, SPELL_MARKED_DEATH, false);
289 events.RepeatEvent(9000);
290 break;
291 }
292
293 DoMeleeAttackIfReady();
294 }

312 else
313 Talk(TEXT_BK_SLAIN_2);
314 }
315 }
316
317 void JustDied(Unit* /*killer*/) override
318 {
319 me->CastSpell((Unit*)nullptr, SPELL_BK_KILL_CREDIT, true);
320 Talk(TEXT_BK_DEATH);

322 pInstance->SetData(BOSS_BLACK_KNIGHT, DONE);

324 me->ToTempSummon()->SetTempSummonType(TEMPSUMMON_MANUAL_DESPAWN);
325 }

427 {
428 switch(spell->Id)
429 {
430 case SPELL_CLAW_N:
431 case SPELL_CLAW_H:
432 DoResetThreat();
433 if (Unit* target = SelectTarget(SelectTargetMethod::Random, 0, 30.0f))
434 {
435 me->AddThreat(target, 100.0f);
436 AttackStart(target);
437 }
438 break;
439 case SPELL_EXPLODE_H:
440 if (target && target->GetTypeId() == TYPEID_PLAYER)
441 if (pInstance)
442 pInstance->SetData(DATA_ACHIEV_IVE_HAD_WORSE, 0);
443 break;
444 }
445 }
446
447 void UpdateAI(uint32 diff) override
448 {
449 if( !UpdateVictim() )
450 return;
451
452 events.Update(diff);
453
454 if( me->HasUnitState(UNIT_STATE_CASTING) )
455 return;
456
457 switch( events.ExecuteEvent() )
458 {
459 case 0:
460 break;
461 case 1: // leap
462 if (Unit* target = me->GetVictim())
463 if (me->GetDistance(target) > 5.0f && me->GetDistance(target) < 30.0f)
464 {
465 me->CastSpell(target, SPELL_LEAP, false);
466
467 break;
468 }
469 events.RepeatEvent(1000);
470 break;
471 case 2: // claw
472 if (Unit* target = me->GetVictim())
473 me->CastSpell(target, SPELL_CLAW_N, false);
474 events.RepeatEvent(urand(6000, 8000));
475 break;
476 }
477
478 DoMeleeAttackIfReady();
479 }
