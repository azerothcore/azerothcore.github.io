181 template<class T> inline void initLengths(T& cacher)
182 {
183 index_type i = index_lo;
184 lengths.resize(index_hi + 1);
185 length_type prev_length = 0, new_length = 0;
186 while (i < index_hi)
187 {
188 new_length = cacher(*this, i);
189 // length overflowed, assign to max positive value
190 if (new_length < 0)
191 new_length = std::numeric_limits<length_type>::max();
192 lengths[++i] = new_length;
193
194 ASSERT(prev_length <= new_length);
195 prev_length = new_length;
196 }
197 }
198
200 [[nodiscard]] length_type length() const { return lengths[index_hi];}
202 [[nodiscard]] length_type length(index_type first, index_type last) const { return lengths[last] - lengths[first];}
203 [[nodiscard]] length_type length(index_type Idx) const { return lengths[Idx];}
204
205 void set_length(index_type i, length_type length) { lengths[i] = length;}
206 void clear();
207 };
